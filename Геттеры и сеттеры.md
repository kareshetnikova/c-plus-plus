## Пример использования геттеров и сеттеров


```
#include <iostream>

using namespace std;

//по принципу инкапсуляции нельзя  оставлять public
//чтобы предоставлять доступ, надо использовать геттеры и сеттеры
//это методы класса для взаимодействия с полями   

class Point
{
public: //геттеры и сеттеры

    //геттер - получаем x
    //int, потому что х интовый
    int GetX()
        {return x;}

    //сеттер - устанавливаем x
    //принимаем valueX в main(), например
    void SetX(int valueX)
        {x=valueX;}

//пример инкапсуляции
private:
    int x;
    int y;

};


int main()
{
    int x1=56;
    int z;
    Point a;
    a.SetX(x1);//сеттер - чтобы установить значение X
    z=a.GetX()+28;//геттер - чтоб в коде получить и использовать значение X
    cout<<"Z="<<z;
    return 0;
}
```
